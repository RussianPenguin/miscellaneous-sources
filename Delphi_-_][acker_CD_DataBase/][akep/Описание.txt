Файл с описанием софта на диске хакера нового образца имеет следующий формат:
Main
{
	Number = [номер в формате dd.mm (№№)]
	Mounth = [Mounth yyyy]
	[Остальный параметры, которые нас мало интересуют]
	[param] = [sense]
}
Block
{
	name = _root
	showlist = [За что отвечает данный параметр я, к сожалению, не знаю]
	Article
	{
		name = [sense] //Обычно в этом разделе содержится интра, поэтому sense = INTRO
		textline = [Собственно сама интра. Таких текстлайнов может быть неограниченное кол-во.]
		[Остальный параметры, которые нас мало интересуют]
		[param] = [sense]
	}
	Block
	{
		name = [Раздел: MAGAZINE, UNIXOID, WINDOWS, ...]
		showlist = 1 //Он везде - 1 :-)
		Block
		{
			name = [Название подраздела: PC ZONE, ВЗЛОМ и др для MAGAZINE, например]
			showlist = 1
			Entry //Самое главное
			{
				name = [имя софтины]
				url = [www софта] //Опциональный параметр
				executable = [путь к софту на диске]
				status = [тип софта (ShareWare, FreeWare и т.д.)] //Опциональный параметр
				author = [размер на диске :-)] //Опциональный параметр
				textline = [Собственно сама интра. Таких текстлайнов может быть неограниченное кол-во.]
			}
		}
		[...] //И т.д. и т.п.
	}
	[...] //И т.д. и т.п.
}
